#### 1.2 Refactorer les backends pour utiliser `serialization.py`

Chaque backend (`json_file.py`, `sqlite.py`, `sqlalchemy.py`) délègue la sérialisation au module centralisé au lieu de dupliquer la logique :

```python
# sqlite.py — avant (dupliqué, cassé)
def _serialize_job(self, job):
    return {"name": job.name, "parameters": job.parameters, ...}

# sqlite.py — après (centralisé, correct)
from ..core.models.serialization import job_to_dict, dict_to_job

def _serialize_job(self, job):
    data = job_to_dict(job)
    # Adaptations spécifiques SQLite (JSON encode des sous-structures)
    data["steps"] = json.dumps(data["steps"])
    data["metadata"] = json.dumps(data["metadata"])
    return data
```

#### 1.3 Créer `tests/integration/test_persistence_roundtrip.py`

```python
@pytest.mark.parametrize("backend_factory", [
    lambda: InMemoryPersistence(),
    lambda tmp: JSONFilePersistence(str(tmp)),
    lambda: SQLitePersistence(":memory:"),
])
def test_job_roundtrip(backend, sample_job):
    backend.save_job(sample_job)
    retrieved = backend.get_job(sample_job.name)
    assert retrieved.name == sample_job.name
    assert len(retrieved.steps) == len(sample_job.steps)

def test_full_lifecycle(backend, sample_job):
    engine = WorkflowEngine(persistence=backend)
    run = engine.run_with_persistence(sample_job)
    retrieved = backend.get_job_run(run.job_run_id)
    assert retrieved.status == run.status
```

#### 3.2 Corriger la détection de signature

```python
# executors/function.py — AVANT
if step.callable.__code__.co_argcount > 0:
    return step.callable(context)

# executors/function.py — APRÈS
import inspect

sig = inspect.signature(step.callable)
params = [
    p for p in sig.parameters.values()
    if p.name not in ("self", "cls")
    and p.kind not in (p.VAR_POSITIONAL, p.VAR_KEYWORD)
]
if params:
    return step.callable(context)
else:
    return step.callable()
```
